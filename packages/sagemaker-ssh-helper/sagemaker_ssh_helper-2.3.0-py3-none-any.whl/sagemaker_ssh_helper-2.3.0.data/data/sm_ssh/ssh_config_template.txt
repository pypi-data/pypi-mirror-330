# *.studio.sagemaker
# *.training.sagemaker
# *.inference.sagemaker
# *.transform.sagemaker
# *.processing.sagemaker
# *.notebook.sagemaker
Host *.*.sagemaker
  IdentityFile ~/.ssh/%h
  PasswordAuthentication no
  ConnectTimeout 120
  ServerAliveInterval 15
  ServerAliveCountMax 8
  ProxyCommand sm-ssh start-proxy %h
  User root
